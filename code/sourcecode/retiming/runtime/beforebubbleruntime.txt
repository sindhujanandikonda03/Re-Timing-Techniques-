/* sort arr[left]...arr[right] into increasing order */
void sortArray(int a[], int length)
{
	int  len=length;
	bool swapped= true;
	while(swapped)
	{
	    swapped=false;
	    for(int i=1;i<len;i++)
	    {
	        int temp=0;
	        if(a[i-1]>a[i])
	        {
	            temp=a[i-1];
	            a[i-1]=a[i];
	            a[i]=temp;
	            swapped=true;
	        }
	    }
	}
  
}
