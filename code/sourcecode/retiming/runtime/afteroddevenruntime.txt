std::list<bool> Sort::q;

void Sort::swap(std::vector<int> &a, int i, int j)
{

		int temp = a[i];

		a[i] = a[j];

		a[j] = temp;

}

void Sort::paf(bool val)
{

q.push_back(val);

}

bool Sort::pif(int a, int b)
{

q.push_back(a > b);

return q.front();

}
